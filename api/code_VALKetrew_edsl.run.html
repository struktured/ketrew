let run t =
  t#activate;
  let rec go_through_deps t =
    t#render :: List.concat_map t#dependencies ~f:go_through_deps in
  let targets =
    go_through_deps t
    |> List.dedup ~compare:Target.(fun ta tb -> compare ta.id tb.id)
  in
  match targets with
  | first :: more -> [`Make (first, more)]
  | [] -> assert false
